### Quick Start - Any Paradigm
```python
# Fractal
from fractal_network import FractalNetwork, FractalConfig
config = FractalConfig(max_depth=8, c_real=-0.7, c_imag=0.27015)

output = network.forward([0.5, -0.3, 0.8, 0.1])

from perpetual_probabilities import PerpetualModel
model = PerpetualModel({'seed': 42, 'specialization': 'language'})

model = QuantumGPT(vocab_size=100, n_embd=32, n_layer=2)

from bio_inspired import EvolutionaryAlgorithm
ga = EvolutionaryAlgorithm(population_size=50)
ga.initialize(genome_size=100, architecture=[4, 8, 4])
ga.evolve(fitness_func, generations=100)

from swarm_intelligence import ParticleSwarmOptimizer
pso = ParticleSwarmOptimizer(n_particles=30, dim=10)
best = pso.optimize(objective_func, iterations=100)

from self_modifying import SelfModifyingSystem
system = SelfModifyingSystem(seed=42)
for _ in range(10):
    system.evolve()

from unified_universe import UnifiedUniverse
universe = UnifiedUniverse(config)
for _ in range(100):
    universe.step()
```
